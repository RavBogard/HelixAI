Create the /api/preview endpoint that runs the full generation pipeline (Planner -> Knowledge Layer) and returns a VisualizerState-shaped response instead of a downloadable file.

    ```typescript
    import type { PresetSpec, BlockSpec, SnapshotSpec, DeviceTarget } from "@/lib/helix/types";

    export interface PreviewResult {
      device: DeviceTarget;
      baseBlocks: BlockSpec[];
      snapshots: [SnapshotSpec, SnapshotSpec, SnapshotSpec, SnapshotSpec];
      presetName: string;
      description: string;
      tempo: number;
    }

    export function hydrateVisualizerState(
      presetSpec: PresetSpec,
      device: DeviceTarget
    ): PreviewResult {
      // baseBlocks = signalChain (already fully parameterized by Knowledge Layer)
      const baseBlocks = presetSpec.signalChain;

      // Always take exactly 4 snapshots for UI (Helix presets have 8, Pod Go has 4)
      // Pad with empty snapshots if fewer than 4
      const snapshots = padSnapshots(presetSpec.snapshots);

      return {
        device,
        baseBlocks,
        snapshots,
        presetName: presetSpec.name,
        description: presetSpec.description,
        tempo: presetSpec.tempo,
      };
    }
    ```

    The `padSnapshots` helper should:
    1. Take the first 4 snapshots from the array
    2. If fewer than 4, create empty SnapshotSpec objects with name "Snapshot N", ledColor 0, empty blockStates {}, empty parameterOverrides {}
    3. Return a tuple of exactly 4
