export const DEVICE_IDS: Record<DeviceTarget, number> = {

The .hlx file format (plain JSON):
```json
{
  "version": 6,
  "data": {
    "device": 2162692,        ← THIS is what must match the connected hardware
    "device_version": 57671680,
    "meta": { ... },
    "tone": { ... }
  },
  "schema": "L6Preset"
}
```

HX Edit checks data.device against the device ID reported by the connected hardware.
If they do not match → error -8309: Incompatible target device type.
