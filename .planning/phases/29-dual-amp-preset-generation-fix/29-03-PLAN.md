```typescript
  // Split block: placed before the first amp (position 0 or 1)
  // Find the position of the first amp block to place split before it
  const firstAmpPos = blocks
    .filter(b => b.type === "amp")
    .reduce((min, b) => Math.min(min, b.position), Infinity);
  const splitPosition = Math.max(0, firstAmpPos);

  dsp.split = {
    "@model": "HD2_SplitAB",
    "@enabled": true,
    "@position": splitPosition,
  };

  // Join block: placed after the last amp/cab position
  // Find the highest position among amp blocks
  const maxAmpPos = blocks
    .filter(b => b.type !== "cab")
    .reduce((max, b) => Math.max(max, b.position), 0);
  const joinPosition = maxAmpPos + 1;

  dsp.join = {
    "@model": "HD2_MergerMixer",
    "@position": joinPosition,
    "A Level": 1.0,
    "B Level": 1.0,
    "A Pan": 0.5,
    "B Pan": 0.5,
  };
}
```

**Change 3: Fix amp-to-cab association for dual-amp**

The current amp-cab association logic in `buildDsp()` (around line 150-158) uses:
```typescript
const associatedCab = blocks.slice(ampIdx + 1).find(b => b.type === "cab")
  || blocks.find(b => b.type === "cab");
```

This finds the "first cab after this amp" which is fragile for dual-amp. For dual-amp, the association must respect `path`: each amp should reference the cab on the SAME path.

Replace the amp-cab association with path-aware logic:
```typescript
if (block.type === "amp") {
  hlxBlock["@bypassvolume"] = 1;

  // Find the cab on the same path as this amp
  let associatedCab: BlockSpec | undefined;
  if (isDualAmp) {
    associatedCab = blocks.find(b => b.type === "cab" && b.path === block.path);
  }
  // Fallback: original logic (first cab after amp, or any cab)
  if (!associatedCab) {
    const ampIdx = blocks.indexOf(block);
    associatedCab = blocks.slice(ampIdx + 1).find(b => b.type === "cab")
      || blocks.find(b => b.type === "cab");
  }

  if (associatedCab) {
    hlxBlock["@cab"] = `cab${cabIndexMap.get(associatedCab) ?? 0}`;
  }
}
```

This ensures: primary amp (path 0) -> cab0, secondary amp (path 1) -> cab1.

```typescript
// Structural validation for dual-amp presets (DUAL-09)
if (isDualAmp) {
  const tone = result.data.tone;

  // Verify topology is AB
  if (tone.global["@topology0"] !== "AB") {
    throw new Error("Dual-amp preset validation failed: @topology0 must be 'AB'");
  }

  // Verify split and join exist
  if (!tone.dsp0.split || tone.dsp0.split["@model"] !== "HD2_SplitAB") {
    throw new Error("Dual-amp preset validation failed: dsp0 must have split block with HD2_SplitAB");
  }
  if (!tone.dsp0.join || tone.dsp0.join["@model"] !== "HD2_MergerMixer") {
    throw new Error("Dual-amp preset validation failed: dsp0 must have join block with HD2_MergerMixer");
  }

  // Verify two amp blocks exist with different paths
  const ampBlocks: Array<{path: number}> = [];
  for (const [key, value] of Object.entries(tone.dsp0)) {
    if (key.startsWith("block") && value && typeof value === "object" && "@type" in value) {
      const blockType = (value as Record<string, unknown>)["@type"];
      if (blockType === getBlockType("amp")) {
        ampBlocks.push({ path: (value as Record<string, unknown>)["@path"] as number });

    throw new Error("Dual-amp preset validation failed: amp blocks must be on different paths (0 and 1)");
  }
}
```
