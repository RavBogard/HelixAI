```typescript
async function loadConversation(convId: string) {
  try {
    const res = await fetch(`/api/conversations/${convId}`)
    if (!res.ok) {
      setError('Failed to load conversation')
      return
    }
    const data = await res.json()

    conversationIdRef.current = convId
    setConversationId(convId)
    isFirstMessageRef.current = false

    if (data.messages && Array.isArray(data.messages)) {
      setMessages(data.messages.map((m: { role: string; content: string }) => ({
        role: m.role as 'user' | 'assistant',
        content: m.content,
      })))
    }

    if (data.device) {
      setSelectedDevice(data.device as 'helix_lt' | 'helix_floor' | 'pod_go')
    }

    if (data.preset_url) {
      setStoredPresetPath(data.preset_url)
    } else {
      setStoredPresetPath(null)
    }

    const lastMsg = data.messages?.[data.messages.length - 1]
    if (lastMsg?.content?.includes('[READY_TO_GENERATE]') || lastMsg?.role === 'assistant') {
      setReadyToGenerate(true)
    }

    // Phase 28: mark as resumed for UXP-03 continuation chips
    setIsResumingConversation(true)
    // Clean URL param — hard refresh will show clean state
    router.replace('/', { scroll: false })
  } catch {
    setError('Failed to load conversation')
  }
}
```

Find the `ensureConversation()` function. After `setConversationId(conv.id)` is called (the success path), add the event dispatch:

```typescript
const ensureConversation = useCallback(async (): Promise<string | null> => {
  if (conversationIdRef.current) return conversationIdRef.current

  try {
    const supabase = createSupabaseBrowserClient()
    const { data: { user } } = await supabase.auth.getUser()
    if (!user || user.is_anonymous) return null
  } catch {
    return null
  }

  try {
    const res = await fetch('/api/conversations', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ device: selectedDevice }),
    })
    if (!res.ok) return null
    const conv = await res.json()
    conversationIdRef.current = conv.id
    setConversationId(conv.id)
    isFirstMessageRef.current = true

    // Phase 28: notify sidebar so it refreshes the conversation list (SIDE-06)
    window.dispatchEvent(new Event('helixai:conversation-created'))

    return conv.id
  } catch {
    return null
  }
}, [selectedDevice])
```

The `helixai:conversation-created` event is already being listened to in ChatSidebar.tsx (Plan 01) — this dispatch closes the loop. ChatSidebar will call `fetchConversations()` which re-fetches `/api/conversations`, and the new conversation (which will have its title set by the auto-title call a few moments later) will appear in the list.

Note: The auto-title fires asynchronously after the first message stream completes. The conversation will briefly appear as "Untitled Session" in the sidebar, then refresh again when the title is set — OR the title will appear on the next full sidebar refresh. This is acceptable. Do NOT add a second `helixai:conversation-created` dispatch after the title PATCH — the title is cosmetic and the sidebar will show it on next open/close cycle.
